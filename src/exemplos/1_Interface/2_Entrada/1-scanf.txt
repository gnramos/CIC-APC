 Algoritmo 1_scanf

Variaveis
 	i : inteiro
 	c : caractere
 	f : real

 Inicio
 	Escreva ("Digite um número inteiro: ")
 	Leia (i)
 	Escreva ("Você digitou____", i) 

 	Escreva ("Digite um número real:")
 	Leia (f)
 	Escreva ("Você digitou____", f)

 	Escreva ("Digite um caractere:")
 	Leia (c)
 	Escreva ("Você digitou____", c)
 Fim
