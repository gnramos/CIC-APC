 Algoritmo UsoDoDo_while


 Variáveis
	N, i : inteiro

 Início
	N <- 10
	i <- 0

	Escreva ("do-while(i < ", N" )")
	i <- 0
	
	Fazer
		Escreva (" ", i+1)
	Enquanto (i < N)
	FimFazer

 Fim
