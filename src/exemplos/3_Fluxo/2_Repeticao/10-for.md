Variáveis
Inteiro: N <- 10, i<-0

Mostra "Para i <- 0, até i < _-1, passo 5", N

Para i <- 0, até i < N-1, passo 5
	Mostra "i = _", i
Fim-Para

Mostra "Para i <-_, i >= 0, passo -5", N

Para i <- N, até i >= 0, passo -5
	Mostra "i = _", i
Fim-Para
