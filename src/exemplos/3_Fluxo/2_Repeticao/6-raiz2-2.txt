 Algoritmo raiz2_2


 Variáveis
	r, n, iteracoes, i : inteiro

 Início

	Escreva ("Qual o valor de n ?")
	Leia (n)

	Se (n < 2) então
		Se (n < 0) Então
			Escreva ("Não sei calcular a raiz quadrada de número negativo.")
		Senão
			Escreva ("A raiz quadrada de ", n é ",n)
		FimSe
	Senão
		Escreva ("Qual sua estimativa inicial para a raiz de ", n" ?")
		Leia (r)
		Escreva ("Quantas iterações?")
		Leia (iteracoes)

		Se (r <= 0 || iteracoes < 0) Então
			Escreva ("Raiz e/ou quantidade de iterações inválida(s).")
		Senão

			Fazer
			r <- (r+(n/r))/2
			i <- i+1
			Enquanto (i < iteracoes)

			FimEnquanto

			Escreva ("Depois de ", iteracoes " tentativas, a aproximação da raiz de ", n" é ", r)
		FimSe
	FimSe
 Fim
