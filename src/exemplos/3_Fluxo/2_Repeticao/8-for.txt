Algoritmo demo_for2

Variaveis
n <- 10, i <- 0 : inteiro

Inicio


Escreva "Para i <- 0, até _-1", n

Para i <- 0, até i < n-1
	Escreva "_", i
Fim-Para

Escreva " i = _", i

Para , até i < n
	Escreva "Teste da condição no início, não deveria ter entrado nesse laço."
Fim-Para

Fim
