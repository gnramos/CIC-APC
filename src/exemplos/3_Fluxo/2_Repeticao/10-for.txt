Algoritmo demo_for

Variáveis
N <- 10, i<-0 : inteiro

Início


Escreva "Para i <- 0, até i < _-1, passo 5", N

Para i <- 0, até i < N-1, passo 5
	Escreva "i = _", i
Fim-Para

Escreva "Para i <-_, i >= 0, passo -5", N

Para i <- N, até i >= 0, passo -5
	Escreva "i = _", i
Fim-Para

Fim
