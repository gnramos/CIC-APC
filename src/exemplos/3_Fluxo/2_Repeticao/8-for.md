Algoritmo demo_for2

Variaveis
Inteiro: n <- 10, i <- 0

Inicio


Mostra "Para i <- 0, até _-1", n

Para i <- 0, até i < n-1
	Mostra "_", i
Fim-Para

Mostra " i = _", i

Para , até i < n
	Mostra "Teste da condição no início, não deveria ter entrado nesse laço."
Fim-Para

Fim
