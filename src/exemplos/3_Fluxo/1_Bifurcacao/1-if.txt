    Se (-1) 
        Escreva ("Se entrou aqui, a condição é \"verdadeira\".")
        Escreva ("Então -1 tem valor lógico \"verdadeiro\".")

        Escreva ("Então -1 tem valor lógico \"falso\".")
    FimSe

    FimSe
