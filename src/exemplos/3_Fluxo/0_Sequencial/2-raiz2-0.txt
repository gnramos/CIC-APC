 * Implementação de código para aproximar a raiz quadrada de
 * um número inteiro em pseudo-código.                       */


 Algoritmo Raiz2-0

 Variáveis
	n, r : inteiro


 Início

	Escreva ("Qual o valor de n? ")
	Leia (n)

	Escreva ("Qual sua estimativa inicial para a raiz de ", n" ?")
	Leia (r)
	Escreva ("r*r = ", r*r)

	r <- (r+(n/r))/2
	Escreva ("r*r = ", r*r)

	r <- (r+(n/r))/2
	Escreva ("r*r = ", r*r)

	Escreva ("Depois de 3 tentativas, a aproximação da raiz de ", n " é ", r".")

 Fim
