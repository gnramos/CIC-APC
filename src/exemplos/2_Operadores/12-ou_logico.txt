 * Exemplo de utilização do operador ||(OU) em pseudo-código. */


 Algoritmo UsoDoOuLogico

 Variáveis
	V, F : inteiro

 Início

	V <- 1
	F <- 0

	Escreva ("|| (ou lógico): ")

	Se (V OU V) Então
		Escreva ("(V || V) é verdadeiro.")
	Senão
		Escreva ("(V || V) é falso.")
	FimSe

	Se (V OU F) Então
		Escreva ("(V || F) é verdadeiro.")
	Senão
		Escreva ("(V || F) é falso.")
	FimSe

	Se (F OU V) Então
		Escreva ("(F || V) é verdadeiro.")
	Senão
		Escreva ("(F || V) é falso.")
	FimSe

	Se (F OU F) Então
		Escreva ("(F || F) é verdadeiro.")
	Senão
		Escreva ("(F || F) é falso.")
	FimSe

 Fim
