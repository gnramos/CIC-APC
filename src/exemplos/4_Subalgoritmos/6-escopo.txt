 Algoritmo EscopoFibonacci


 Variáveis
	contador, n : inteiro

 
 Função (inteiro) fib(inteiro n)
	contador <- contador + 1
	Se (n < 2) Então
		Retorna 1
	 Retorna fib(n-1) + fib(n-2)
	FimSe
 Fim_Função



 Função (Vazio) testa_fib(inteiro n)
	contador <- 0
	Escreva ("fib(", n") = ", fib(n))
	Escreva ("( ", contador" chamadas)")
 Fim_Função


 Início
