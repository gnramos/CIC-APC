 Algoritmo 0_Estatica


 Variáveis
	bytes, inteiros[50] : inteiro
		f, reais[5] : float
		c, string   : caractere

 Início
	bytes <- sizeof(bytes) + sizeof(inteiros)
	bytes <- bytes + sizeof(f) + sizeof(reais)
	bytes <- bytes + sizeof(c) + sizeof(string)

	Escreva ("Alocação de ", bytes" bytes de memória.)

 Fim
