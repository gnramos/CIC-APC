 Algoritmo 2_automatica


 Definições
	alguem[20]  caractere

 Variáveis
	bytes <- 0: inteiro global


 Função (inteiro) fat(inteiro n)
	Escreva ("Alocação de ", sizeof(n)" bytes de memória (fat).")
	bytes <- bytes + sizeof(n)

	Retorna (n > 1 ? n*fat(n-1) : n)

 Fim_Função

 
 Função (Vazio) F(caractere ponteiro nome, inteiro n)
	Escreva ("Alocação de ", sizeof(n) + size(nome)" bytes de memória (F).")
	bytes <- bytes + sizeof(n) + sizeof(nome)
	
	Escreva ("Olá ", nome " .")
	Escreva (, n " ! = ", fat(n) " .")

 Fim_Função

 
 Início
	alguem[20] <- "Fulano" /* alocação estática*/
	
	F(alguem, 3)

 Fim
