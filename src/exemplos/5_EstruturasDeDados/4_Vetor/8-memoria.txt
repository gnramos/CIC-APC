 Algoritmo Memoria

 Variáveis
	buffer : caractere
	buffer[50], buffer[51] : vetor de caractere
	
 Função (Vazio) scanf_errado()
 	
	//ponteiro buffer

	Escreva "Digite uma frase: "
	Leia(buffer)
	Escreva "Você digitou [", buffer" ]"

 FimFunção


 Função (Vazio) scanf_vulneravel()

	//buffer[50];

	Escreva "Digite uma frase: "
	Leia(buffer)
	Escreva "Você digitou [", buffer" ]"
 
 FimFunção


 Função (Vazio) scanf_seguro()
	
	//buffer[51];
	
	Escreva "Digite uma frase: "
	Leia("%50[^\n]", buffer)
	Escreva "Você digitou [", buffer" ]"
 
 FimFunção
