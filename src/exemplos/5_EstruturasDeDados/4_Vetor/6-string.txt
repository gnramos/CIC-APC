 Algoritmo UsoDeString2

 Constantes
	TAM : inteiro <- 50

 Variáveis

	Frase[TAM] : vetor de caractere

 Início
	Escreva "Digite um string: "
	Leia ("%[^\n]", frase)
	
	mostra_indices(TAM, "          ")

	/* Assume-se que o string termina
  	   em '.'. */

 	mostra_ate_char(frase, '.')

 Fim
