 Algoritmo 2_Ordem

 Definições
	data_t : registro (dia,mes,ano : inteiro)

	aniversario_t : registro (nome[50] : caractere
				  data_t data)
 Variáveis
	i, j, em_ordem : inteiro
	n : inteiro constante
	aniversarios : vetor[10] de aniversario_t

 Função (Vazio) mostra_data(data_t ponteiro data)
	Escreva "%02d/%02d/%04d" % data->dia , % data->mes , % data->ano

 FimFunção

 
 Função (Vazio) mostra(aniversario_t ponteiro aniversario)
	Escreva "%12s", aniversario->nome
	mostra_data(&(aniversario->data))

 FimFunção


 Função (inteiro) alfabetica(aniversario_t a, aniversario_t b)
	Retorna strncmp(a.nome, b.nome, 50) < 0
 
 FimFunção


 Função (inteiro) cronologica(aniversario_t a, aniversario_t b)
	Se (a.data.ano != b.data.ano) Então
		Retorna a.data.ano < b.data.ano
	FimSe
	
	Se (a.data.mes != b.data.mes) Então
		Retorna (a.data.mes < b.data.mes)
	FimSe
	
	Retorna a.data.dia <= b.data.dia

 FimFunção


 Função (inteiro) alfabetica_cronologica(aniversario_t a, aniversario_t b )
	em_ordem <- alfabetica(a,b)
	Retorna em_ordem ? em_ordem : cronologica(a,b)

 FimFunção


 Função (Vazio) troca(aniversario_t ponteiro a, aniversario_t ponteiro b)
	aniversario_t aux <- (ponteiro a)
	(ponteiro a) <- (ponteiro b)
	(ponteiro b) <- aux
 
 FimFunção

 
 Função (Vazio) ordena(aniversario_t ponteiro vetor, inteiro n, inteiro (ponteiro em_ordem)(aniversario_t, aniversario_t))
	Para i de 0 a n Faça
		Para j de i+1 a n Faça
			Se (!em_ordem(vetor[i], vetor[j])) Então
				troca(vetor+i, vetor+j)
			FimSe
		FimPara
	FimPara

 FimFunção


 Início

	n <- 10
	aniversarios[10] <- ({"Charlie Zeta",1, 1, 1980},
        		     {"Alice", 1, 1, 1988},
        		     {"Bob 1", 1, 1, 2000},
        		     {"Alice", 1, 1, 2009},

       		             {"Bob 3", 1, 7, 2000},
       			     {"Bob Silva", 12, 1, 2000},
       			     {"Alice", 1,1, 1999},
        		     {"Charlie Alfa", 1, 1, 1980},
        		     {"Charlie Beta",1, 1, 1980})
	
	ordena(aniversarios, n, cronologica);
    	Escreva "Ordem Cronológica"
    	Para i de 0 a n Faça

	        mostra(aniversarios + i);
	FimPara
    
	ordena(aniversarios, n, alfabetica);
    	Escreva "Ordem Alfabética"
    	Para i de 0 a n Faça
	
        	mostra(aniversarios + i);
	FimPara
	
    	ordena(aniversarios, n, alfabetica_cronologica);
    	Escreva "Ordem Alfabética/Cronológica"
    	Para i de 0 a n Faça
        	
		mostra(aniversarios + i);
	FimPara

 Fim
