 Algoritmo IEEE754

 
 Variáveis
	   bits : inteiro
	bitmask : inteiro sem sinal
	      f : float

 Função (Vazio) print_bits(inteiro n)
	bits <- 8*sizeof(int)
	bitmask <- 0x80000000

	Para bits passo -1 , bitmask <- bitmask/2 Faça
		putchar(n & bitmask ? '1' : '0')
	FimPara
 FimFunção

 Função (Vazio) mostra_bits_IEEE754(inteiro n)
	memcpy(&f,&n, sizeof(f)) /* copia os bits de n em f */

	Escreva "%f : ", f
	print_bits(n)
	putchar('\n')

 FimFunção
